# .github/workflows/release.yaml
name: Release

on:
  release:
    types: [published]

permissions:
  contents: write

jobs:
  release:
    name: Release
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Create zip file
        run: |
          cd custom_components/reolink_license_plate
          zip -r ../../reolink_license_plate.zip .
          
      - name: Upload zip to release
        uses: svenstaro/upload-release-action@v2
        with:
          repo_token: ${{ secrets.GITHUB_TOKEN }}
          file: reolink_license_plate.zip
          asset_name: reolink_license_plate.zip
          tag: ${{ github.ref }}
          overwrite: true

---
# .github/workflows/test.yaml
name: Test

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  test:
    runs-on: ubuntu-latest
    strategy:
      matrix:
        python-version: [3.11, 3.12]

    steps:
      - uses: actions/checkout@v4
      
      - name: Set up Python ${{ matrix.python-version }}
        uses: actions/setup-python@v4
        with:
          python-version: ${{ matrix.python-version }}
          
      - name: Install dependencies
        run: |
          python -m pip install --upgrade pip
          pip install -r requirements_dev.txt
          
      - name: Lint with flake8
        run: |
          flake8 custom_components/reolink_license_plate --count --select=E9,F63,F7,F82 --show-source --statistics
          flake8 custom_components/reolink_license_plate --count --exit-zero --max-complexity=10 --max-line-length=127 --statistics
          
      - name: Test with pytest
        run: |
          pytest tests/ -v
